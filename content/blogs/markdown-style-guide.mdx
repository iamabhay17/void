```tsx
import { useState } from "react";

interface CounterProps {
  initialValue?: number;
}

export function Counter({ initialValue = 0 }: CounterProps) {
  const [count, setCount] = useState(initialValue);

  return (
    <div className="flex items-center gap-4">
      <button onClick={() => setCount((c) => c - 1)}>-</button>
      <span>{count}</span>
      <button onClick={() => setCount((c) => c + 1)}>+</button>
    </div>
  );
}
```

## Long Code Block

This tests scrolling behavior for longer code:

```typescript
// A more complex example to test horizontal scrolling
export class DataProcessor<T extends Record<string, unknown>> {
  private readonly data: T[];
  private readonly transformers: Array<(item: T) => T>;

  constructor(initialData: T[] = []) {
    this.data = initialData;
    this.transformers = [];
  }

  addTransformer(transformer: (item: T) => T): this {
    this.transformers.push(transformer);
    return this;
  }

  process(): T[] {
    return this.data.map((item) =>
      this.transformers.reduce((acc, transformer) => transformer(acc), item),
    );
  }

  filter(predicate: (item: T) => boolean): T[] {
    return this.process().filter(predicate);
  }

  async processAsync(): Promise<T[]> {
    return new Promise((resolve) => {
      setTimeout(() => resolve(this.process()), 100);
    });
  }
}

// Usage example with very long lines to test horizontal scrolling in the code block component
const processor = new DataProcessor([
  {
    id: 1,
    name: "Item 1",
    description:
      "A very long description that might cause horizontal scrolling in the code block",
  },
]);
```
